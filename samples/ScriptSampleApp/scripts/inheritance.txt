class Animal
{
    constructor(name)
	{ 
	    this.name = name;
	}
	name;
    move(meters)
	{
        print(this.name + " moved " + meters + "m.");
    }
}

class Snake extends Animal
{
    constructor(name)
	{ 
	    super(name); 
	}
    move()
	{
        print("Slithering...");
        super.move(5);
    }
}

class Horse extends Animal
{
    constructor(name)
	{ 
	    super(name); 
	}
    move()
	{
        print("Galloping...");
        super.move(45);
    }
}

var sam = new Snake("Sammy the Python");
var tom = new Horse("Tommy the Palomino");

sam.move();                             //  Slithering...
						                //  Sammy the Python moved 5m.
tom.move();								//  Galloping...
										//  Tommy the Palomino moved 45m.
